	text={\textsc{IA}},

\newglossaryentry{enigma}{type=glo,
	text={\textsc{Enigma}},
	name={Enigma},
	description={Machine servant au chiffrement et déchiffrement de messages, d'informations principalement employée par les Allemands}}

\newglossaryentry{alphago}{type=glo,
	text={\textsc{AlphaGo}},
	name={AlphaGo},
	description={Programme informatique qui a pour but de jouer au Go, il a remporté de nombreux prix face à de grands joueurs humains}}

\newglossaryentry{resneurone}{type=glo,
	text={\textsc{Réseau Neuronal}},
	name={Réseau Neuronal},

	text={\textsc{dataset}},

	description={Ensemble de données. Peut être des images, des vidéos, du texte, ...}}

\newglossaryentry{pixel}{type=glo,
	text={\textsc{pixel}},
	name={Pixel},
	description={Unité élémentaire composant une image, possède une unique couleur}}

\newglossaryentry{MNIST}{type=glo,
	text={\textsc{MNIST}},
	name={MNIST Dataset},
	description={Ensemble de données du Mixed National Institute of Standards and Technology, un dataset contenant des chiffres écrits à la main}}

\newglossaryentry{CNN}{type=glo,
	text={\textsc{CNN}},
	name={Convolutional Neural Network},
	description={Réseau neuronal dans lequel le motif de connexion entre les neurones est inspiré du cortex visuel des animaux}}

	text={\textsc{R-CNN}},

\newglossaryentry{TPU}{type=glo,

\newglossaryentry{framework}{type=glo,

\newglossaryentry{npl}{type=glo,
	text={\textsc{Natural Processing Language}},
	name={Natural Processing Language},
	description={Ensemble de techniques qui ont pour but d'harmoniser le langage humain et le langage machine}}

\newglossaryentry{appnonsuper}{type=glo,
	text={\textsc{apprentissage non-supervisé}},
	name={Apprentissage non-supervisé},

\newglossaryentry{apprenfor}{type=glo,
	text={\textsc{apprentissage par renforcement}},
	name={Apprentissage par renforcement},
	description={Apprentissage par sélection ou récompense. On récompense le réseau neuronal pour les réponses les plus adaptées qu'il donne}}

\newglossaryentry{deep}{type=glo,
	text={\textsc{deep learning}},
	name={Deep Learning},
	description={Apprentissage profond. Devenu un terme générique pour parler d'un algorithme possèdant un réseau neuronal}}

\newglossaryentry{transfer}{type=glo,
	text={\textsc{transfer learning}},
	name={Transfer Learning},
	description={Concept qui prend un réseau neuronal pré-entraîné et qui l'exploite sur une application similaire sans passer par la phase d'apprentissage exclusif}}
